import streamlit as st import torch import torch.nn as nn import torch.optim as optim import os import time # --- 1. ARQUITECTURA DE LA IA --- class RedNeuronal(nn.Module): def __init__(self, entrada_dim=3, oculto_dim=16, salida_dim=1): super(RedNeuronal, self).__init__() self.capa1 = nn.Linear(entrada_dim, oculto_dim) self.relu = nn.ReLU() self.capa2 = nn.Linear(oculto_dim, salida_dim) def forward(self, x): x = self.capa1(x) x = self.relu(x) x = self.capa2(x) return x # --- 2. LÓGICA DE ENTRENAMIENTO --- def entrenar_modelo_base(): """Entrena un modelo rápido si no existe uno guardado.""" INPUT_SIZE, HIDDEN_SIZE, OUTPUT_SIZE = 3, 16, 1 modelo = RedNeuronal(INPUT_SIZE, HIDDEN_SIZE, OUTPUT_SIZE) criterio = nn.MSELoss() optimizador = optim.Adam(modelo.parameters(), lr=0.01) # Datos sintéticos rápidos X = torch.randn(100, INPUT_SIZE) y = X.sum(dim=1, keepdim=True) for _ in range(500): optimizador.zero_grad() perdida = criterio(modelo(X), y) perdida.backward() optimizador.step() torch.save(modelo.state_dict(), "modelo_ia.pth") return modelo # --- 3. CONFIGURACIÓN DE LA INTERFAZ (ESTILO GEMINI) --- st.set_page_config(page_title="Mi IA Chat", page_icon="⚡", layout="centered") # CSS para personalizar el estilo y que se vea más moderno st.markdown(""" """, unsafe_allow_html=True) st.title("IA Neuronal Chat 🤖") st.caption("Interfaz impulsada por PyTorch y Streamlit") # --- 4. GESTIÓN DEL MODELO Y SESIÓN --- if "modelo" not in st.session_state: if not os.path.exists("modelo_ia.pth"): with st.spinner("Entrenando cerebro de la IA por primera vez..."): st.session_state.modelo = entrenar_modelo_base() else: modelo = RedNeuronal() modelo.load_state_dict(torch.load("modelo_ia.pth")) modelo.eval() st.session_state.modelo = modelo if "messages" not in st.session_state: st.session_state.messages = [ {"role": "assistant", "content": "¡Hola! Soy tu IA. Envíame 3 números separados por comas y predeciré el resultado."} ] # --- 5. RENDERIZADO DEL CHAT --- for message in st.session_state.messages: with st.chat_message(message["role"]): st.markdown(message["content"]) # Entrada del usuario if prompt := st.chat_input("Ejemplo: 0.5, 1.2, -0.8"): # Mostrar mensaje del usuario st.session_state.messages.append({"role": "user", "content": prompt}) with st.chat_message("user"): st.markdown(prompt) # Respuesta de la IA with st.chat_message("assistant"): placeholder = st.empty() full_res = "" try: # Procesar entrada nums = [float(x.strip()) for x in prompt.split(",")] if len(nums) != 3: respuesta = "Necesito exactamente **3 números** para procesar la información." else: # Inferencia tensor_in = torch.tensor([nums], dtype=torch.float32) with torch.no_grad(): pred = st.session_state.modelo(tensor_in).item() respuesta = f"Mi red neuronal procesó tus datos. El resultado estimado es: **{pred:.4f}**" except ValueError: respuesta = "Por favor, asegúrate de enviar solo números separados por comas." # Efecto de escritura for palabra in respuesta.split(): full_res += palabra + " " time.sleep(0.08) placeholder.markdown(full_res + "▌") placeholder.markdown(full_res) st.session_state.messages.append({"role": "assistant", "content": full_res})
